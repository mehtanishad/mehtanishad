def read_file_contents(file_path):
    data = []
    with open(file_path, 'r') as file:
        for line in file:
            line = line.strip()
            pairs = line.split(',')
            dictionary = {}
            for pair in pairs:
                key, value = pair.split(':')
                dictionary[key] = int(value)
                data.append(dictionary)
    return data
